@views.route('/', methods=['GET', 'POST', 'DELETE'])
@login_required
def home():
 if request.method == "POST":
  numpi = request.form["numpi"]
  fibs = [0, 1]
  while fibs[-1] < int(numpi):
   fibs.append(fibs[-1] + fibs[-2])
  if int(numpi) in fibs:
   flash('O número {} está presente na sequência de Fibonacci'.format(numpi), category='success')
   new_numpi = ("O número {} foi encontrado na sequência de Fibonacci".format(numpi))
  else:
   flash('O número {} não está presente na sequência de Fibonacci!'.format(numpi), category='error')
   new_numpi = ("O número {} não foi encontrado na sequência de Fibonacci!".format(numpi))
 return render_template(new_numpi=new_numpi)
